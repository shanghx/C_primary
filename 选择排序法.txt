#include<iostream>
using namespace std;
int main()
{
	int i,j,a[100],temp,n,k;
	cout<<"input数字的个数"<<endl;
	cin>>n; 
	for(i=0;i<n;i++)
	cin>>a[i];
	for(i=0;i<n-1;i++)
	{
	k=i;
	for(j=i+1;j<n;j++)
	if(a[k]>a[j]) k=j;
	{
		temp=a[i];
		a[i]=a[k];
		a[k]=temp;
	 };
	 }
	for(i=0;i<n;i++)
	cout<<a[i]<<" ";
	 return 0;
}
